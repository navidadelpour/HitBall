========================================================== GAME DESIGN ==========================================================
balls {
    after killing some enemies you special ability activated,
    every ball has a different amount of speed and height,
    ball features {
        more health,
        random item for a time,
        destroing obstacles,
        using enemies as score,
        spiky ball,
        gunner ball,
    }
}

combo system {
    near,
    enemies killing,
}

obstacles {
    obstacles on air(roof),
    closing door from top,
    falling objects from top(or bottom),
    arrows from right,
}

items {
    speed_increaser,
    flyer,
    electrical_shock,
    underground,
    scaler,
    jump_power,
}

other features {
    adding roof,
    adding challenges like not hitting the coins or coils or ...,
    gun,
}

shop {
    item slot,
    item upgrading(increasing time or do it more times),
    some items are unlockable,
    upgrading speed increase amount,
    upgrading height increase amount,
    balls,
    theme,
    guns,
}
